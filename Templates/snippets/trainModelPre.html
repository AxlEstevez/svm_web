def vectorizar(X_train, X_test):
    tfidf = TfidfVectorizer(stop_words='english')
    X_train_numeros=tfidf.fit_transform(X_train)
    X_test_numeros = tfidf.transform(X_test)

    return X_train_numeros, X_test_numeros, tfidf
